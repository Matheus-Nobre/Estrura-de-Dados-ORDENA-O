#include <iostream>
using namespace std;

int main(){
    
    int n;
    
    cin >> n;
    
    int A[n];
    
    for(int i = 0 ; i < n; i++){
        cin>> A[i];
    }
    
    int aux, count = 0;
    bool swapped = true;
    
    while(swapped){
    //for(int = 0; i < n; i++){
        swapped = false;
        count++;
        
        for(int j = 0; j < n-1; j++ ){
            if(A[j] > A[j+1]){
                aux = A[j];
                A[j] = A[j + 1];
                A[j + 1] = aux;
                swapped = true;
            }
        }
    }
    
    cout<< count<< endl;
  
  
    return 0;
}
