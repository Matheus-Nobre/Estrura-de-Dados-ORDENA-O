#include <iostream>
using namespace std;

int main(){
    
    int n;
    
    cin >> n;
    
    int A[n];
    int B[n];
    
    for(int i = 0 ; i < n; i++){
        cin>> A[i];
    }
    
    for(int i = 0 ; i < n; i++){
        cin>> B[i];
    }
    
    int aux;
    bool swapped = true;
    
    while(swapped){
        swapped = false;
        
        for(int j = 0; j < n-1; j++ ){
            if(A[j] > A[j+1]){
                aux = A[j];
                A[j] = A[j + 1];
                A[j + 1] = aux;
                swapped = true;
            }
        }
    }
    
    swapped = true;
    while(swapped){
        swapped = false;
        
        for(int j = 0; j < n-1; j++ ){
            if(B[j] > B[j+1]){
                aux = B[j];
                B[j] = B[j + 1];
                B[j + 1] = aux;
                swapped = true;
            }
        }
    }
    
    
    int cont = 0;
    for (int i = 0; i < n; i++) {
        if(A[i] < B[i]){
            cout<< "No" << endl; 
            break;
        }
        cont++;
    }
    if (cont == n){
        cout<< "Yes" << endl;
    }
    
    
    return 0;
}
