#include <iostream>
#include <vector>

using namespace std;

void indices(vector<int> vetor){
    vector<int> aux = vetor;
    bool swapped = true;
    int n = vetor.size();
    while(swapped != false){
        swapped = false;
        for (int i = 0; i < n-1; i++)
            if(vetor[i] > vetor[i+1]){
                swap(vetor[i],vetor[i+1]);
                swapped = true;
            }
    }
    for(int i = 0; i < n;i++){
        for (int j = 0; j < n; j++)
            if(vetor[i] == aux[j]){
                cout<<j;
                break;
            }
        if(i != n-1)
            cout<<" ";
    }
}
int main(){
    int n = 0;
    cin>>n;
    vector<int> vetor;
    for(int i = 0;i < n;i++){
        int num;
        cin>>num;
        vetor.push_back(num);
    }
    indices(vetor);
    cout<<"\n";
    return 0;
}
