#include <iostream>
using namespace std;

int main(){
    
    int n;
    
    cin >> n;
    
    int A[n];
    
    for(int i = 0 ; i < n; i++){
        cin>> A[i];
    }
    
    int aux;
    bool swapped = true;
    
    while(swapped){
        swapped = false;
        
        for(int j = 0; j < n-1; j++ ){
            if(A[j] > A[j+1]){
                aux = A[j];
                A[j] = A[j + 1];
                A[j + 1] = aux;
                swapped = true;
            }
        }
    }
    
    int cont = 0;
    for (int i = 0; i < n; i++) {
        if(A[i] == A[i+1]){
            cont++;
        }
    }
    
    cout << cont<< endl;
    return 0;
}
